134 void BattleBot::react(BattleBot* attacker, json attackData)
135 {
136  if(attacker == this)
137  {
138  mAttacks++;
139  return;
140  }
141 
142  if(attacker == nullptr)
143  {
144  return;
145  }
146 
147  if(mHealth == 0)
148  {
149  return;
150  }
151 
152  bool criticalHit = attackData["critical hit"].get<bool>();
153  bool criticalMiss = attackData["critical miss"].get<bool>();
154 
155  std::string attackLog = attackData["attacker"].get<std::string>();
156 
157  if(criticalMiss)
158  {
159  attackLog += " (critically misses) automatically failing the attack.";
160  }
161  else if (criticalHit)
162  {
163  attackLog += " [critically hits] dealing additional damage.";
164  }
165  else
166  {
167  attackLog += " attacks " + name() + " with a {" + std::to_string(attackData["attack"].get<int32_t>()) + "}.";
168  }
169 
170  // Check to see if the attack beat our AC
171  if((attackData["attack"].get<int32_t>() < AC() && !criticalHit) || criticalMiss)
172  {
173  mAttacksBlocked++;
174  attackLog += " " + name() + " (blocks).";
175  mRunner.log(attackLog);
176  }
177  else
178  {
179  auto damage = std::max<int32_t>(attackData["damage"].get<int32_t>(),1);
180 
181  attackLog += " " + name() + " takes [" + std::to_string(damage) + "] damage.";
182  mRunner.log(attackLog);
183 
184  attacker->mHits++;
185  attacker->mDamageDealt += damage;
186 
187  // They hurt us
188  if(mHealth <= damage)
189  {
190  // We died
191  mHealth = 0;
192  mRunner.log(name() + " [dies].");
193 
194  attacker->mKills++;
195  }
196  else
197  {
198  mHealth -= damage;
199 
200  }
201  }
202 }

391 void BattleBot::attack()
392 {
393  json attackData;
394 
395  for(auto i = 0; i < num_attacks(); ++i)
396  {
397  attackData["attacker"] = name();
398 
399  auto roll = mRunner.roll(1, 20, attack_modifier());
400  auto diceToRoll = num_damage_die();
401 
402  attackData["critical miss"] = false;
403  attackData["critical hit"] = false;
404  if(roll == 1+attack_modifier())
405  {
406  attackData["critical miss"] = true;
407  diceToRoll = 0;
408  }
409  else if(roll == 20+attack_modifier())
410  {
411  attackData["critical hit"] = true;
412  diceToRoll *= 2;
413  }
414 
415  attackData["attack"] = roll;
416  attackData["damage"] = mRunner.roll(diceToRoll, damage_die(), damage_modifier());
417 
418  mRunner.react_bots(this, attackData);
419  }
420 }
